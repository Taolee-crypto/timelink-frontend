// API 설정 - 백엔드 URL 수정 const API_BASE_URL = 'https://timelink-backend.timelink-api.workers.dev'; // DOM 로드 시 실행 document.addEventListener('DOMContentLoaded', async function() { console.log('TimeLink 대시보드 로드됨'); // 인증 확인 (6자리 승인번호 방식에 맞게 수정) const isAuthenticated = await checkAuthentication(); if (!isAuthenticated) { window.location.href = 'login.html'; return; } // 사용자 정보 로드 await loadUserData(); // TL 잔액 업데이트 (백엔드에 해당 API가 없으므로 임시) updateBalanceDisplay(); // 페이지 주기적 업데이트 setInterval(updateBalanceDisplay, 30000); }); // 인증 확인 (6자리 승인번호 방식) async function checkAuthentication() { // localStorage에서 인증 상태 확인 const verificationStatus = localStorage.getItem('verification_status'); const userEmail = localStorage.getItem('user_email'); const authCode = localStorage.getItem('auth_code'); console.log('인증 상태 확인:', { verificationStatus, userEmail, authCode: authCode ? '있음' : '없음' }); // 1. 인증이 완료된 상태인지 확인 if (verificationStatus === 'verified' && userEmail) { console.log('이미 인증된 사용자:', userEmail); return true; } // 2. 인증코드가 있고, 이메일이 있으면 서버에 인증 확인 if (authCode && userEmail) { console.log('인증코드로 서버 인증 시도'); const isVerified = await verifyWithServer(userEmail, authCode); if (isVerified) { localStorage.setItem('verification_status', 'verified'); console.log('서버 인증 성공'); return true; } else { console.log('서버 인증 실패'); return false; } } // 3. 아무런 인증 정보도 없음 console.log('인증 정보 없음 - 로그인 페이지로 이동'); return false; } // 서버에 인증 확인 요청 async function verifyWithServer(email, code) { try { console.log('서버 인증 요청:', { email, code }); const response = await fetch(`${API_BASE_URL}/api/auth/verify`, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ email: email, code: code }) }); console.log('서버 응답 상태:', response.status); if (response.ok) { const result = await response.json(); console.log('인증 성공:', result); // 인증 성공 시 사용자 정보 저장 if (result.user) { localStorage.setItem('user_data', JSON.stringify(result.user)); } return true; } else { console.error('인증 실패:', response.status); return false; } } catch (error) { console.error('인증 요청 오류:', error); // 개발 모드: 네트워크 오류 시에도 로컬 정보로 진행 if (process.env.NODE_ENV === 'development') { console.warn('개발 모드: 로컬 인증 사용'); localStorage.setItem('verification_status', 'verified'); return true; } return false; } } // 사용자 정보 로드 (백엔드 API 없으므로 로컬 데이터 사용) async function loadUserData() { try { // 로컬 스토리지에서 사용자 정보 가져오기 let userData = JSON.parse(localStorage.getItem('user_data') || '{}'); const userEmail = localStorage.getItem('user_email') || 'user@example.com'; console.log('로컬 사용자 데이터:', userData); // 기본 사용자 데이터 설정 if (!userData.name) { userData = { name: userEmail.split('@')[0] || '사용자', email: userEmail, balance: 10000, // 기본 보너스 10,000 TL role: 'user', created_at: new Date().toISOString() }; localStorage.setItem('user_data', JSON.stringify(userData)); } // 사용자 이름 표시 if (userData.name) { document.getElementById('userName').textContent = userData.name; document.getElementById('welcomeTitle').textContent = `${userData.name}님, 환영합니다!`; // 아바타 초기 설정 (이름 첫 글자) const firstLetter = userData.name.charAt(0).toUpperCase(); document.getElementById('userAvatar').textContent = firstLetter; } // TL 잔액 표시 updateBalanceDisplay(); } catch (error) { console.error('사용자 정보 로드 오류:', error); // 오류 시 기본값 설정 document.getElementById('userName').textContent = '사용자'; document.getElementById('welcomeTitle').textContent = '환영합니다!'; document.getElementById('userAvatar').textContent = 'U'; } } // 잔액 표시 업데이트 (백엔드 API가 없으므로 로컬 데이터 사용) function updateBalanceDisplay() { try { const userData = JSON.parse(localStorage.getItem('user_data') || '{}'); const balance = userData.balance || 0; console.log('잔액 업데이트:', balance); // UI 업데이트 if (document.getElementById('userBalance')) { document.getElementById('userBalance').textContent = balance.toLocaleString(); } if (document.getElementById('mainBalance')) { document.getElementById('mainBalance').textContent = `${balance.toLocaleString()} TL`; } } catch (error) { console.error('잔액 업데이트 오류:', error); } } // 로그아웃 처리 async function handleLogout(event) { if (event) event.preventDefault(); console.log('로그아웃 처리 시작'); // 로컬 스토리지 정리 (인증 정보만 제거) localStorage.removeItem('verification_status'); localStorage.removeItem('auth_code'); localStorage.removeItem('user_email'); localStorage.removeItem('remember_me'); // 로그인 페이지로 이동 console.log('로그아웃 완료 - 로그인 페이지로 이동'); window.location.href = 'login.html'; }
